// ── Persistent Log (writes to "Logs" sheet) ─────────────────────────
function logToSheet(source, message, details) {
  try {
    var ss = SpreadsheetApp.openById(CONFIG.QA_SHEET_ID);
    var logSheet = ss.getSheetByName('Logs');
    if (!logSheet) {
      logSheet = ss.insertSheet('Logs');
      logSheet.appendRow(['timestamp', 'source', 'message', 'details']);
      logSheet.setColumnWidth(1, 180);
      logSheet.setColumnWidth(2, 100);
      logSheet.setColumnWidth(3, 300);
      logSheet.setColumnWidth(4, 500);
    }
    logSheet.appendRow([
      new Date().toISOString(),
      source || 'system',
      String(message || '').substring(0, 500),
      String(details || '').substring(0, 1000)
    ]);
    // Keep only last 200 rows to prevent sheet bloat
    var lastRow = logSheet.getLastRow();
    if (lastRow > 210) {
      logSheet.deleteRows(2, lastRow - 201);
    }
  } catch (e) {
    Logger.log('logToSheet error: ' + e.message);
  }
}

// ── CORS + Routing ──────────────────────────────────────────────────
function doGet(e) {
  var p = e.parameter || {};
  var action = p.action || '';
  var result;

  switch (action) {
    case 'telegram_status':
      result = telegramStatus();
      break;
    case 'poll':
      result = pollAnswer(p.id || '');
      break;
    case 'qa_knowledge':
      result = getQAKnowledge();
      break;
    case 'health':
      result = { status: 'ok', engine: 'Google Apps Script', qa_count: getQACount(), version: '2.0' };
      break;
    case 'ai':
      result = askAI(p.question || '', p.context || '');
      break;
    case 'telegram':
      result = askTelegram(p.question || '', p.parent_name || 'Phụ huynh', p.context || '');
      break;
    case 'debug_poll':
      result = debugPollStatus();
      break;
    case 'force_poll':
      result = forcePoll();
      break;
    case 'load_edits':
      result = loadRoadmapEdits();
      break;
    case 'save_edits':
      result = saveRoadmapEditsJSON(p.data || '{}');
      break;
    case 'lookup_employee':
      result = lookupEmployee(p.msnv || '');
      break;
    case 'get_roleplay_sessions':
      result = getRoleplaySessions(p.employee_id || '', p.bu || '', p.limit || '50');
      break;
    case 'set_offset':
      if (p.offset) {
        PropertiesService.getScriptProperties().setProperty('lastUpdateId', p.offset);
        logToSheet('setup', 'lastUpdateId set via web', p.offset);
        result = { ok: true, lastUpdateId: p.offset };
      } else {
        result = { error: 'Missing offset param' };
      }
      break;
    default:
      result = { error: 'Unknown action', version: '2.0' };
  }

  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var body = {};
  try {
    body = JSON.parse(e.postData.contents);
  } catch (err) {
    // Maybe form data
  }

  var p = e.parameter || {};
  var action = p.action || body.action || '';
  var result;

  switch (action) {
    case 'ai':
      result = askAI(body.question || p.question || '', body.context || p.context || '');
      break;
    case 'telegram':
      result = askTelegram(body.question || p.question || '', body.parent_name || p.parent_name || 'Phụ huynh', body.context || p.context || '');
      break;
    case 'webhook':
      result = handleTelegramWebhook(body);
      break;
    case 'log_roleplay':
      result = logRoleplaySession(body);
      break;
    default:
      return doGet(e);
  }

  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

// ── AI Expert (Perplexity sonar-pro + RAG) ──────────────────────────
function askAI(question, context) {
  if (!question) return { error: 'No question provided' };

  var systemPrompt = buildSystemPrompt(question, context);

  var payload = {
    model: 'sonar-pro',
    messages: [
      { role: 'system', content: systemPrompt },
      { role: 'user', content: question },
    ],
    max_tokens: 1024,
    temperature: 0.5,
    search_recency_filter: 'month',
  };

  try {
    var resp = UrlFetchApp.fetch('https://api.perplexity.ai/chat/completions', {
      method: 'post',
      contentType: 'application/json',
      headers: { 'Authorization': 'Bearer ' + CONFIG.PERPLEXITY_API_KEY },
      payload: JSON.stringify(payload),
      muteHttpExceptions: true,
    });

    var data = JSON.parse(resp.getContentText());
    if (data.choices && data.choices[0]) {
      return { answer: data.choices[0].message.content, mode: 'ai', model: 'sonar-pro' };
    }
    return { error: 'Invalid API response', detail: resp.getContentText().substring(0, 200) };
  } catch (err) {
    return { error: 'AI request failed', detail: err.message };
  }
}

// ── Process a single Telegram update (shared by webhook + polling) ──
function processOneUpdate(update) {
  var msg = update.message || {};
  var replyTo = msg.reply_to_message || {};
  var replyMsgId = replyTo.message_id;
  var text = msg.text;

  if (!replyMsgId || !text) return { ok: true, action: 'ignored', reason: 'no_reply_or_text' };

  // Dedup: skip if we already processed this message_id
  var msgId = msg.message_id;
  if (msgId) {
    var props = PropertiesService.getScriptProperties();
    var processedKey = 'processed_msg_' + msgId;
    if (props.getProperty(processedKey)) {
      return { ok: true, action: 'dedup_skip', message_id: msgId };
    }
    // Mark as processed (store for 24h via expiry check)
    props.setProperty(processedKey, String(new Date().getTime()));
  }

  try {
    var ss = SpreadsheetApp.openById(CONFIG.QA_SHEET_ID);
    var questionsSheet = ss.getSheetByName('Questions');
    if (!questionsSheet) return { ok: true, action: 'no_questions_sheet' };

    var data = questionsSheet.getDataRange().getValues();
    var matchRow = -1;
    var questionId = '';
    var question = '';
    var context = '';

    for (var i = 1; i < data.length; i++) {
      // Column E (index 4) = telegram_msg_id — compare as numbers
      if (Number(data[i][4]) === Number(replyMsgId)) {
        matchRow = i + 1; // 1-indexed for Sheet API
        questionId = data[i][0];
        question = data[i][1];
        context = data[i][3];
        break;
      }
    }

    if (matchRow === -1) {
      logToSheet('processUpdate', 'No match for replyMsgId=' + replyMsgId, 
        'Available msgIds: ' + data.slice(1).map(function(r){ return r[4]; }).join(','));
      return { ok: true, action: 'no_match', replyMsgId: replyMsgId };
    }

    // Get sender name
    var sender = msg.from || {};
    var name = sender.first_name || 'Chuyên gia';
    if (sender.last_name) name += ' ' + sender.last_name;

    // 1. Save to Answers sheet (for polling)
    var answersSheet = ss.getSheetByName('Answers');
    if (!answersSheet) {
      answersSheet = ss.insertSheet('Answers');
      answersSheet.appendRow(['question_id', 'text', 'name', 'timestamp']);
    }
    answersSheet.appendRow([questionId, text, name, new Date().toISOString()]);

    // 2. Update Questions status
    questionsSheet.getRange(matchRow, 7).setValue('answered');

    // 3. Save to QA Knowledge (main sheet)
    var qaSheet = ss.getSheetByName('Trang tính1') || ss.getSheets()[0];
    var qaData = qaSheet.getDataRange().getValues();
    var existingRow = -1;
    for (var j = 1; j < qaData.length; j++) {
      if (qaData[j][0] === question) { existingRow = j + 1; break; }
    }
    if (existingRow > 0) {
      var cur = qaSheet.getRange(existingRow, 2).getValue();
      var curBy = qaSheet.getRange(existingRow, 3).getValue();
      qaSheet.getRange(existingRow, 2).setValue(cur + '\n---\n' + name + ': ' + text);
      qaSheet.getRange(existingRow, 3).setValue(curBy + ', ' + name);
      qaSheet.getRange(existingRow, 5).setValue(new Date().toISOString());
    } else {
      qaSheet.appendRow([question, name + ': ' + text, name, context, new Date().toISOString()]);
    }

    logToSheet('processUpdate', 'SAVED answer for ' + questionId, 'by=' + name + ' text=' + text.substring(0, 100));
    return { ok: true, action: 'saved', question_id: questionId, answered_by: name };

  } catch (e) {
    logToSheet('processUpdate', 'ERROR: ' + e.message, e.stack);
    return { ok: false, error: e.message };
  }
}

// ── Telegram Polling (runs every minute via trigger) ─────────────────
// BACKUP: Polling via trigger (optional, no longer needed since pollAnswer checks inline)
// Keep for backward compatibility but can be removed.
function pollTelegramReplies() {
  var props = PropertiesService.getScriptProperties();
  var lastUpdateId = parseInt(props.getProperty('lastUpdateId') || '0');

  try {
    var url = 'https://api.telegram.org/bot' + CONFIG.TELEGRAM_BOT_TOKEN + '/getUpdates';
    var resp = UrlFetchApp.fetch(url, {
      method: 'post',
      contentType: 'application/json',
      payload: JSON.stringify({ offset: lastUpdateId + 1, timeout: 0 }),
      muteHttpExceptions: true,
    });

    var responseText = resp.getContentText();
    var data = JSON.parse(responseText);

    if (!data.ok) {
      // Only log errors that aren't the usual "no updates" case
      if (responseText.indexOf('409') > -1) {
        // 409 conflict — another getUpdates is running, just skip this cycle
        return;
      }
      logToSheet('poll', 'Telegram API error', responseText.substring(0, 500));
      return;
    }

    if (!data.result || data.result.length === 0) {
      return; // No updates, stay quiet
    }

    logToSheet('poll', 'Got ' + data.result.length + ' updates', 
      JSON.stringify(data.result.map(function(u) {
        var m = u.message || {};
        var rt = m.reply_to_message || {};
        return {
          uid: u.update_id,
          text: (m.text || '').substring(0, 50),
          reply_to: rt.message_id || null,
          from: (m.from || {}).first_name || '?'
        };
      })).substring(0, 900));

    // Process each update independently
    var newMaxId = lastUpdateId;
    for (var idx = 0; idx < data.result.length; idx++) {
      var update = data.result[idx];
      newMaxId = Math.max(newMaxId, update.update_id);

      try {
        var saveResult = processOneUpdate(update);
        if (saveResult.action === 'saved') {
          logToSheet('poll', 'SAVED ' + (saveResult.question_id || ''), 'by=' + (saveResult.answered_by || ''));
        }
      } catch (e) {
        logToSheet('poll', 'Error update ' + update.update_id, e.message);
      }
    }

    // Confirm offset AFTER all processing
    props.setProperty('lastUpdateId', String(newMaxId));

  } catch (e) {
    logToSheet('poll', 'FATAL', e.message);
  }
}

// ── Debug endpoint: check polling status ────────────────────────────
function debugPollStatus() {
  var props = PropertiesService.getScriptProperties();
  var lastUpdateId = props.getProperty('lastUpdateId') || '0';

  // Get recent logs
  var logs = [];
  try {
    var ss = SpreadsheetApp.openById(CONFIG.QA_SHEET_ID);
    var logSheet = ss.getSheetByName('Logs');
    if (logSheet) {
      var lastRow = logSheet.getLastRow();
      var startRow = Math.max(2, lastRow - 19); // last 20 logs
      if (lastRow >= 2) {
        var logData = logSheet.getRange(startRow, 1, lastRow - startRow + 1, 4).getValues();
        logs = logData.map(function(r) {
          return { time: r[0], source: r[1], msg: r[2], detail: r[3] };
        });
      }
    }
  } catch (e) {
    logs = [{ error: e.message }];
  }

  // Check Telegram pending
  var pending = 0;
  try {
    var resp = UrlFetchApp.fetch(
      'https://api.telegram.org/bot' + CONFIG.TELEGRAM_BOT_TOKEN + '/getWebhookInfo',
      { muteHttpExceptions: true }
    );
    var info = JSON.parse(resp.getContentText());
    pending = info.result ? info.result.pending_update_count : 0;
  } catch (e) {}

  // Check triggers
  var triggers = ScriptApp.getProjectTriggers().map(function(t) {
    return {
      func: t.getHandlerFunction(),
      type: String(t.getEventType()),
      id: t.getUniqueId()
    };
  });

  return {
    version: '2.0',
    lastUpdateId: lastUpdateId,
    telegram_pending: pending,
    triggers: triggers,
    recent_logs: logs,
    timestamp: new Date().toISOString()
  };
}

  pollTelegramReplies();
  return debugPollStatus();
}

// ── Poll for answers (frontend polling) ─────────────────────────────
// KEY DESIGN: Every time frontend polls, we also check Telegram for new
// replies inline. This way we don't depend on time-based triggers.
function pollAnswer(questionId) {
  if (!questionId) return { status: 'error', detail: 'No question_id' };

  // Step 1: Actively check Telegram for new replies
  try {
    checkTelegramUpdates();
  } catch (e) {
    // Non-fatal: continue checking sheet even if Telegram fails
  }

  // Step 2: Check Answers sheet
  var ss = SpreadsheetApp.openById(CONFIG.QA_SHEET_ID);
  var answersSheet = ss.getSheetByName('Answers');
  if (!answersSheet) return { status: 'waiting' };

  var data = answersSheet.getDataRange().getValues();
  var answers = [];
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] === questionId) {
      answers.push({
        text: data[i][1],
        name: data[i][2],
        time: data[i][3],
      });
    }
  }

  if (answers.length > 0) {
    return {
      status: 'answered',
      answer: answers[answers.length - 1].text,
      answered_by: answers[answers.length - 1].name,
      answered_at: answers[answers.length - 1].time,
      answers: answers,
      answer_count: answers.length,
    };
  }

  // Check if question exists and timeout
  var questionsSheet = ss.getSheetByName('Questions');
  if (questionsSheet) {
    var qData = questionsSheet.getDataRange().getValues();
    for (var i = 1; i < qData.length; i++) {
      if (qData[i][0] === questionId) {
        var askedAt = new Date(qData[i][5]);
        var elapsed = (new Date() - askedAt) / 1000;
        if (elapsed > 1800) return { status: 'timeout' };
        break;
      }
    }
  }

  return { status: 'waiting' };
}

// ── Check Telegram for new replies (called inline from pollAnswer) ───
function checkTelegramUpdates() {
  var props = PropertiesService.getScriptProperties();
  var lastUpdateId = parseInt(props.getProperty('lastUpdateId') || '0');

  var resp = UrlFetchApp.fetch(
    'https://api.telegram.org/bot' + CONFIG.TELEGRAM_BOT_TOKEN + '/getUpdates',
    {
      method: 'post',
      contentType: 'application/json',
      payload: JSON.stringify({ offset: lastUpdateId + 1, timeout: 0 }),
      muteHttpExceptions: true,
    }
  );

  var data = JSON.parse(resp.getContentText());
  if (!data.ok || !data.result || data.result.length === 0) return;

  var newMaxId = lastUpdateId;
  for (var idx = 0; idx < data.result.length; idx++) {
    var update = data.result[idx];
    newMaxId = Math.max(newMaxId, update.update_id);
    try {
      var result = processOneUpdate(update);
      if (result.action === 'saved') {
        logToSheet('inline', 'SAVED ' + (result.question_id || ''), 'by=' + (result.answered_by || ''));
      }
    } catch (e) {
      logToSheet('inline', 'Error', e.message);
    }
  }

  props.setProperty('lastUpdateId', String(newMaxId));
}

// ── Q&A Knowledge Base ──────────────────────────────────────────────
function getQAKnowledge() {
  var ss = SpreadsheetApp.openById(CONFIG.QA_SHEET_ID);
  var sheet = ss.getSheetByName('Trang tính1') || ss.getSheets()[0];
  var data = sheet.getDataRange().getValues();
  var pairs = [];
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] && data[i][1]) {
      pairs.push({ q: data[i][0], a: data[i][1], by: data[i][2], context: data[i][3], time: data[i][4] });
    }
  }
  return { count: pairs.length, pairs: pairs };
}

function getQACount() {
  try {
    var ss = SpreadsheetApp.openById(CONFIG.QA_SHEET_ID);
    var sheet = ss.getSheetByName('Trang tính1') || ss.getSheets()[0];
    return Math.max(0, sheet.getLastRow() - 1);
  } catch (e) {
    return 0;
  }
}

function retrieveQAKnowledge(question) {
  var ss = SpreadsheetApp.openById(CONFIG.QA_SHEET_ID);
  var sheet = ss.getSheetByName('Trang tính1') || ss.getSheets()[0];
  var data = sheet.getDataRange().getValues();
  if (data.length <= 1) return '';

  var queryWords = question.toLowerCase().split(/\s+/);
  var scored = [];

  for (var i = 1; i < data.length; i++) {
    if (!data[i][0] || !data[i][1]) continue;
    var entryWords = (data[i][0] + ' ' + data[i][1]).toLowerCase().split(/\s+/);
    var entrySet = {};
    entryWords.forEach(function(w) { entrySet[w] = true; });
    var overlap = 0;
    queryWords.forEach(function(w) { if (entrySet[w]) overlap++; });
    if (overlap >= 2) scored.push({ score: overlap, q: data[i][0], a: data[i][1], by: data[i][2] });
  }

  scored.sort(function(a, b) { return b.score - a.score; });
  var top = scored.slice(0, 5);
  if (top.length === 0) return '(Chưa có Q&A từ Ban lãnh đạo)';

  return top.map(function(e) { return 'Q: ' + e.q + '\nA (' + e.by + '): ' + e.a; }).join('\n\n');
}

// Run this ONCE to start polling every 1 minute
function setupPollingTrigger() {
  // Remove all existing pollTelegramReplies triggers
  ScriptApp.getProjectTriggers().forEach(function(t) {
    if (t.getHandlerFunction() === 'pollTelegramReplies') ScriptApp.deleteTrigger(t);

// ── Cleanup old dedup keys (run occasionally) ───────────────────────
function cleanupDedupKeys() {
  var props = PropertiesService.getScriptProperties();
  var all = props.getProperties();
  var now = new Date().getTime();
  var cutoff = now - 24 * 60 * 60 * 1000; // 24 hours
  var removed = 0;
  for (var key in all) {
    if (key.indexOf('processed_msg_') === 0) {
      var ts = parseInt(all[key]);
      if (ts < cutoff) {
        props.deleteProperty(key);
        removed++;
      }
    }
  }
  logToSheet('cleanup', 'Removed ' + removed + ' old dedup keys', '');
}

function loadRoadmapEdits() {
  try {
    var sheet = getEditsSheet();
    var data = sheet.getDataRange().getValues();
    var result = {};
    for (var i = 1; i < data.length; i++) {
      if (data[i][0]) result[data[i][0]] = data[i][1];
    }
    return result;
  } catch (e) {
    return { error: e.message };
  }
}

function saveRoadmapEditsJSON(jsonStr) {
  try {
    var edits = JSON.parse(jsonStr);
    var sheet = getEditsSheet();
    var data = sheet.getDataRange().getValues();
    var existingKeys = {};
    for (var i = 1; i < data.length; i++) {
      if (data[i][0]) existingKeys[data[i][0]] = i + 1;
    }
    var saved = 0;
    var now = new Date().toISOString();
    var keys = Object.keys(edits);
    for (var idx = 0; idx < keys.length; idx++) {
      var k = keys[idx];
      var v = edits[k];
      if (existingKeys[k]) {
        var cell = sheet.getRange(existingKeys[k], 2);
        cell.setNumberFormat("@");
        cell.setValue(v);
        sheet.getRange(existingKeys[k], 3).setValue(now);
      } else {
        var newRow = sheet.getLastRow() + 1;
        sheet.appendRow([k, '', now]);
        var newCell = sheet.getRange(newRow, 2);
        newCell.setNumberFormat("@");
        newCell.setValue(v);
      }
      saved++;
    }
    return { ok: true, saved: saved };
  } catch (e) {
    return { error: e.message };
  }
}

// ── Employee Lookup (from "Nvien" sheet) ────────────────────────────
function lookupEmployee(msnv) {
  if (!msnv) return { ok: false, error: 'Vui lòng nhập Mã NV' };
  try {
    var ss = SpreadsheetApp.openById(CONFIG.QA_SHEET_ID);
    var sheet = ss.getSheetByName('Nvien');
    if (!sheet) return { ok: false, error: 'Không tìm thấy danh sách nhân viên' };
    var data = sheet.getDataRange().getValues();
    var headers = data[0];
    var colMSNV = headers.indexOf('MSNV');
    var colName = headers.indexOf('Ho ten');
    var colPosition = headers.indexOf('Vi tri');
    var colBU = headers.indexOf('Co so (BU)');
    var colRegion = headers.indexOf('Khu vuc');
    var colStatus = headers.indexOf('Trang thai');

    if (colMSNV < 0) return { ok: false, error: 'Cấu trúc sheet nhân viên không đúng' };

    var searchVal = msnv.trim().toUpperCase();
    for (var i = 1; i < data.length; i++) {
      if (String(data[i][colMSNV]).trim().toUpperCase() === searchVal) {
        return {
          ok: true,
          employee: {
            msnv: data[i][colMSNV],
            name: colName >= 0 ? data[i][colName] : '',
            position: colPosition >= 0 ? data[i][colPosition] : '',
            bu: colBU >= 0 ? data[i][colBU] : '',
            region: colRegion >= 0 ? data[i][colRegion] : '',
            status: colStatus >= 0 ? data[i][colStatus] : ''
          }
        };
      }
    }
    return { ok: false, error: 'Không tìm thấy mã nhân viên: ' + msnv };
  } catch (e) {
    logToSheet('employee_lookup', 'Error: ' + e.message, msnv);
    return { ok: false, error: 'Lỗi hệ thống, vui lòng thử lại' };
  }
}

// ── Get Role-Play Sessions (from "RolePlaySessions" sheet) ──────────
function getRoleplaySessions(employeeId, bu, limitStr) {
  try {
    var ss = SpreadsheetApp.openById(CONFIG.QA_SHEET_ID);
    var sheet = ss.getSheetByName('RolePlaySessions');
    if (!sheet) return { ok: true, sessions: [], total: 0 };

    var data = sheet.getDataRange().getValues();
    if (data.length <= 1) return { ok: true, sessions: [], total: 0 };

    var headers = data[0];
    var limit = parseInt(limitStr) || 50;
    var sessions = [];

    for (var i = data.length - 1; i >= 1 && sessions.length < limit; i--) {
      var row = {};
      for (var j = 0; j < headers.length; j++) {
        row[headers[j]] = data[i][j];
      }
      if (employeeId && row.employeeId !== employeeId) continue;
      if (bu && row.BU !== bu) continue;
      // Skip heavy fields for list view
      delete row.rawTranscriptJson;
      delete row.rawScoreJson;
      sessions.push(row);
    }

    return { ok: true, sessions: sessions, total: sessions.length };
  } catch (e) {
    return { ok: false, error: 'Lỗi đọc sessions', sessions: [] };
  }
}
